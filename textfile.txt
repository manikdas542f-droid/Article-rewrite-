<script>
const synonyms = {
    "good": ["great","excellent","fine","wonderful"],
    "bad": ["poor","terrible","awful"],
    "big": ["large","huge","massive"],
    "small": ["tiny","little","compact"],
    "important": ["crucial","vital","key"],
    "use": ["utilize","apply","employ"],
    "make": ["create","build","develop"],
    "get": ["obtain","receive","gain"]
};

function getRandomSynonym(word) {
    const w = word.toLowerCase();
    return synonyms[w]
        ? synonyms[w][Math.floor(Math.random() * synonyms[w].length)]
        : word;
}

function rewriteSentence(sentence) {
    return sentence.split(/\s+/).map(word => {
        const match = word.match(/([a-zA-Z]+)([^a-zA-Z]*)/);
        return match ? getRandomSynonym(match[1]) + match[2] : word;
    }).join(' ');
}

function rewriteArticle() {
    const input = document.getElementById('inputText').value.trim();
    if (!input) {
        document.getElementById('output').innerText = "Please enter text!";
        return;
    }

    const paragraphs = input.split('\n').filter(p => p.trim());
    const rewritten = paragraphs.map(p => {
        const sentences = p.split(/(?<=[.!?])\s+/);
        return sentences.map(s => rewriteSentence(s)).join(' ');
    });

    document.getElementById('output').innerText = rewritten.join('\n\n');
}
